  <button id="connectBtn">Connect with Stripe</button>
  <div id="status"></div>

  <script>
    const connectBtn = document.getElementById('connectBtn');
    const statusDiv = document.getElementById('status');

    connectBtn.addEventListener('click', async () => {
      statusDiv.textContent = "Redirecting to Stripe...";
      connectBtn.disabled = true;

      try {
        // Call your Firebase function to create a Stripe Connect account link
        const response = await fetch('/__/functions/createConnectAccount'); // This is your Firebase function URL
        const data = await response.json();

        if (data.url) {
          // Redirect user to Stripe onboarding
          window.location.href = data.url;
        } else {
          statusDiv.textContent = "Failed to create Stripe account link. Try again.";
          connectBtn.disabled = false;
        }
      } catch (err) {
        console.error(err);
        statusDiv.textContent = "Error connecting to Stripe. Check console.";
        connectBtn.disabled = false;
      }
    });
  </script>
